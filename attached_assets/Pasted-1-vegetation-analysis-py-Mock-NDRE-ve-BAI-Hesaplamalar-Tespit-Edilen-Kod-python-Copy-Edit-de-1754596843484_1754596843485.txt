def calculate_bai(self):
    """Calculate BAI (Burn Area Index)"""
    b, g, r = cv2.split(self.image.astype(np.float32))
    # Mock BAI calculation
    bai = 1.0 / ((0.1 - r)**2 + (0.06 - g)**2 + 1e-8)
    return np.clip(bai, 0, 100)
